# Dev-просмотрщик с реальным графиком — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Дать backend отдавать данные ряда в `chart.data` и поднять страницу `GET /ui`, рисующую реальный график (Chart.js) по ответу `/api/chat`.

**Architecture:** Аддитивное расширение `ChartSpec` полем `data` (заполняет агент `viz`, усечение по `MAX_CHART_POINTS`, приведение DuckDB-типов к JSON-safe). Новый route `GET /ui` отдаёт self-contained HTML-страницу с Chart.js из CDN, которая бьёт в боевой `POST /api/chat`.

**Tech Stack:** Python, FastAPI, Pydantic, DuckDB, pytest + httpx (ASGITransport), Chart.js (CDN).

Заменить содержимое `app/agents/viz.py` целиком на:

```python
import time
from datetime import date, datetime
from decimal import Decimal
from app.config import settings
from app.contracts import PipelineState, ChartSpec, TraceStep

_TIME_HINTS = ("month", "quarter", "year", "date", "day")

def _looks_temporal(name: str) -> bool:
    n = name.lower()
    return any(h in n for h in _TIME_HINTS)

def _json_safe(v):
    if isinstance(v, Decimal):
        return float(v)
    if isinstance(v, (date, datetime)):
        return v.isoformat()
    return v

def _chart_data(rows: list[dict]) -> list[dict]:
    return [{k: _json_safe(val) for k, val in r.items()}
            for r in rows[: settings.max_chart_points]]

def visualize(state: PipelineState) -> PipelineState:
    t0 = time.monotonic()
    ex = state.extraction
    cols, rows = (ex.columns, ex.rows) if ex else ([], [])

    if not rows or len(rows) == 1:
        # одно число ИЛИ одна строка: график из единственной точки бессмыслен
        spec = ChartSpec(type="none", reason="одна строка/одно число — график не нужен")
    elif len(cols) >= 2 and _looks_temporal(cols[0]):
        spec = ChartSpec(type="line", x=cols[0], y=cols[1],
                         series=cols[2] if len(cols) > 2 else None,
                         reason="динамика во времени")
    elif len(cols) >= 2:
        spec = ChartSpec(type="bar", x=cols[0], y=cols[1],
                         reason="сравнение по категориям")
    else:
        spec = ChartSpec(type="none", reason="график не уместен")

    if spec.type != "none":
        spec.data = _chart_data(rows)

    state.chart = spec
    state.trace.append(TraceStep(agent="viz", summary=f"{spec.type}: {spec.reason}",
                                 elapsed_ms=int((time.monotonic() - t0) * 1000)))
    return state
```

let chartInstance = null;

async function ask() {
  const msg = document.getElementById('msg').value;
  const sid = document.getElementById('sid').value.trim();
  const status = document.getElementById('status');
  status.textContent = 'запрос…';
  const body = { message: msg };
  if (sid) body.session_id = sid;
  let data;
  try {
    const r = await fetch('/api/chat', {
      method: 'POST',
      headers: { 'content-type': 'application/json' },
      body: JSON.stringify(body),
    });
    data = await r.json();
    status.textContent = 'HTTP ' + r.status;
  } catch (e) {
    status.textContent = 'ошибка сети: ' + e;
    return;
  }
  render(data);
}

function render(d) {
  document.getElementById('resp').textContent = d.response || '(пусто)';
  const insuf = document.getElementById('insuf');
  insuf.className = 'badge ' + (d.insufficient_data ? 'warn' : 'ok');
  insuf.textContent = d.insufficient_data ? 'данных недостаточно' : 'ответ';

  const ul = document.getElementById('assum');
  ul.innerHTML = '';
  (d.assumptions || []).forEach(a => {
    const li = document.createElement('li'); li.textContent = a; ul.appendChild(li);
  });

  const tb = document.querySelector('#trace tbody');
  tb.innerHTML = '';
  (d.trace || []).forEach(t => {
    const tr = document.createElement('tr');
    tr.innerHTML = `<td>${t.agent||''}</td><td>${t.summary||''}</td>`
      + `<td>${t.rows??''}</td><td>${t.elapsed_ms??''}</td>`
      + `<td class="sql">${t.sql? t.sql.replace(/</g,'&lt;') : ''}</td>`;
    tb.appendChild(tr);
  });

  renderChart(d.chart);
}

function renderChart(chart) {
  const note = document.getElementById('chartnote');
  if (chartInstance) { chartInstance.destroy(); chartInstance = null; }
  if (typeof Chart === 'undefined') { note.textContent = 'Chart.js недоступен (нет сети до CDN)'; return; }
  if (!chart || chart.type === 'none' || !chart.data || !chart.data.length) {
    note.textContent = chart ? ('графика нет: ' + (chart.reason||'')) : 'графика нет';
    return;
  }
  note.textContent = 'тип: ' + chart.type + ' · X=' + chart.x + ' · Y=' + chart.y
    + (chart.series ? ' · series=' + chart.series : '') + ' · ' + (chart.reason||'');
  const ctx = document.getElementById('chart').getContext('2d');
  const rows = chart.data;
  let cfg;

  if (chart.type === 'scatter') {
    cfg = { type: 'scatter', data: { datasets: [{ label: chart.y,
      data: rows.map(r => ({ x: r[chart.x], y: r[chart.y] })) }] } };
  } else if (chart.type === 'grouped_bar' && chart.series) {
    const labels = [...new Set(rows.map(r => r[chart.x]))];
    const groups = [...new Set(rows.map(r => r[chart.series]))];
    cfg = { type: 'bar', data: { labels, datasets: groups.map(g => ({
      label: g,
      data: labels.map(l => {
        const hit = rows.find(r => r[chart.x] === l && r[chart.series] === g);
        return hit ? hit[chart.y] : 0;
      }),
    })) } };
  } else {
    const kind = (chart.type === 'line' || chart.type === 'area') ? 'line' : 'bar';
    cfg = { type: kind, data: {
      labels: rows.map(r => r[chart.x]),
      datasets: [{ label: chart.y, data: rows.map(r => r[chart.y]),
        fill: chart.type === 'area' }],
    } };
  }
  chartInstance = new Chart(ctx, cfg);
}

document.getElementById('ask').addEventListener('click', ask);
</script>

</html>
